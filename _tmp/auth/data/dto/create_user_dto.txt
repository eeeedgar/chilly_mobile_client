import 'package:freezed_annotation/freezed_annotation.dart';

@JsonSerializable()
class CreateUserDto {
  final String name;
  final String login;
  final String profilePicUrl;
  final String password;
  final List<Role> role;

  CreateUserDto(

  Map<String, dynamic> toJson() => _$CreateUserDtoToJson(this);
}
