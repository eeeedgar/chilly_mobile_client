import 'package:chilly_mobile_client/features/auth/data/dto/create_user_dto.txt';
import 'package:chilly_mobile_client/features/register/data/login_dto.dart';
import 'package:dio/dio.dart';

class AuthRepository {
  final dio = Dio();
  final basePath = 'https://chilly-39na.onrender.com/api';

  Future<void> register(CreateUserDto dto) async {
    await dio
        .post(
          '$basePath/register',
          data: dto.toJson(),
        )
        .then((response) => print('[createEvent] $response'));
  }

  Future<void> login(LoginDto dto) async {
    await dio
        .put(
          '$basePath/login',
          data: dto.toJson(),
        )
        .then((response) => print('[createEvent] $response'));
  }
}

class UserInfo {
  static String? name;
  static String? login;
  static String? profilePicUrl;
  static String? id;
  static DateTime? registeredAt;
  static DateTime? updatedAt;
  static String? password;
  static List<String>? roles;
}
