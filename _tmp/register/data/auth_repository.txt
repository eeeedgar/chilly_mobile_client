import 'package:chilly_mobile_client/features/auth/data/dto/create_user_dto.txt';
import 'package:dio/dio.dart';

class AuthRepository {
  final dio = Dio();
  final basePath = 'https://chilly-39na.onrender.com/api';

  Future<void> createUser(CreateUserDto dto) async {
    await dio
        .post(
          '$basePath/register',
          data: dto.toJson(),
        )
        .then((response) => print('[createUser] $response'));
  }
}
